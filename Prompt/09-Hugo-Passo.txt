Since I cannot execute SQL directly on your Supabase instance currently, please run the following script in your Supabase SQL Editor to enable all features:

sql
-- 1. Add new columns to estimates
ALTER TABLE public.estimates 
ADD COLUMN IF NOT EXISTS status TEXT DEFAULT 'Draft',
ADD COLUMN IF NOT EXISTS project_type TEXT,
ADD COLUMN IF NOT EXISTS amount_paid NUMERIC(10,2) DEFAULT 0,
ADD COLUMN IF NOT EXISTS balance_due NUMERIC(10,2) DEFAULT 0,
ADD COLUMN IF NOT EXISTS payment_status TEXT DEFAULT 'unpaid',
ADD COLUMN IF NOT EXISTS public_token UUID DEFAULT gen_random_uuid(),
ADD COLUMN IF NOT EXISTS sent_at TIMESTAMPTZ,
ADD COLUMN IF NOT EXISTS approved_at TIMESTAMPTZ;
-- 2. Create index for status filtering
CREATE INDEX IF NOT EXISTS idx_estimates_org_status ON public.estimates(organization_id, status);
-- 3. Create Payments Table
CREATE TABLE IF NOT EXISTS public.estimate_payments (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id uuid REFERENCES public.organizations(id) NOT NULL,
  estimate_id uuid REFERENCES public.estimates(id) ON DELETE CASCADE NOT NULL,
  amount numeric(10,2) NOT NULL,
  payment_method text CHECK (payment_method IN ('cash', 'check', 'card', 'bank_transfer')),
  payment_date timestamptz DEFAULT now(),
  created_at timestamptz DEFAULT now()
);
-- 4. Enable RLS and Policies for payments
ALTER TABLE public.estimate_payments ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can view payments of their organization" 
ON public.estimate_payments FOR SELECT 
USING (organization_id IN (SELECT organization_id FROM public.profiles WHERE id = auth.uid()));
CREATE POLICY "Users can insert payments for their organization" 
ON public.estimate_payments FOR INSERT 
WITH CHECK (organization_id IN (SELECT organization_id FROM public.profiles WHERE id = auth.uid()));
-- 5. Final verification of existing policies
ALTER TABLE public.company_settings ENABLE ROW LEVEL SECURITY;
